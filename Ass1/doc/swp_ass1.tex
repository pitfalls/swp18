object SimpleParserA extends JavaTokenParsers {
  def A: Parser[Any] = "aa" ~ A | "c" ~ ""
  def B: Parser[Any] = "b" ~ B | "c" ~ ""
  def AB: Parser[Any] = A | B

  def S: Parser[Any] = "a" ~ AB | "ac"
  def apply(s: String) = parseAll(S, s)
}
\end{lstlisting}
